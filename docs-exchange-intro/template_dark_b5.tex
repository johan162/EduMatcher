% --- SHORT TOC ---
% Command: \shorttoc{Title}{Depth}
% Depth 0 = Chapters, 1 = Chapters + Sections, etc.
\newcommand{\shorttocentry}[1]{%
  {\makeatletter
  \renewcommand{\numberline}[1]{\makebox[4.5em][l]{##1}}%
  \MakeUppercase{#1}%
  \makeatother}%
}
{%
  \renewcommand{\contentsname}{Chapters}%
  % Reduce top whitespace on short TOC page so all chapter entries fit.
  \titlespacing*{\chapter}{0pt}{0pt}{0.8em}%
  \makeatletter
  \renewcommand{\l@chapter}[2]{%
    \addvspace{5pt}%
    \noindent{%
      \sffamily\bfseries\large\color{brandblue}\shorttocentry{#1}%
      \hfill{\normalsize\mdseries\color{brandgray}#2}%
    }\par%
  }%
  \makeatother
  \shorttoc{Chapters}{0}%
}
\endgroup
